DROP TABLE accion;
DROP TABLE  empresa;
DROP TABLE  usuario;


CREATE TABLE usuario(
   cedula numeric(10),
   contraseña varchar(10),
   titular varchar(10),
   telefono numeric(10),
   email varchar(10),
   numCuenta numeric(10),
   saldo numeric(10),
   fechaRegistro date,
   PRIMARY KEY(cedula)
);


CREATE TABLE Historial(
   origen numeric(10),
   destino numeric(10),
   Monto numeric(10),
   fecha date
);

CREATE TABLE empresa(
   idEmpresa NUMERIC(10),
   nombreEmpresa VARCHAR(20),
   cantAcciones NUMERIC(10),
   cantAccionesAct NUMERIC(10),
   precioAccion NUMERIC(20),
   PRIMARY KEY (idEmpresa)
);

CREATE TABLE accion(
   idEmpresa NUMERIC(10),
   cantidad NUMERIC(10),
   ceduladueño NUMERIC(10),
   PRIMARY KEY (idEmpresa,ceduladueño),
   FOREIGN KEY (idEmpresa) REFERENCES empresa(idEmpresa),
   FOREIGN KEY (ceduladueño) REFERENCES usuario(cedula)
);
